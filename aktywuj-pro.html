(function(){ const rank = { basic:1, pro:2, elite:3 }; function getPlan(){ try { return JSON.parse(localStorage.getItem('qm_subscription_v1')||'null')?.plan || localStorage.getItem('qm_plan_v1') || 'basic'; } catch(e){ return 'basic'; } } function allow(required){ return rank[getPlan()] >= rank[required]; } document.addEventListener('DOMContentLoaded',()=>{ document.querySelectorAll('[data-require]').forEach(el=>{ const required = el.getAttribute('data-require'); if (!allow(required)) { el.classList.add('locked'); if (el.tagName === 'A') { el.setAttribute('href','cennik.html'); el.setAttribute('title', 'Wymagany plan ' + required.toUpperCase()); } if (el.tagName === 'BUTTON') { el.disabled = false; el.addEventListener('click', ()=> location.href = 'cennik.html'); } if (!el.querySelector('.qm-lock-label')) { const span = document.createElement('span'); span.className = 'qm-lock-label'; span.style.marginLeft = '8px'; span.style.fontWeight = '800'; span.textContent = '🔒 ' + required.toUpperCase(); el.appendChild(span); } } }); }); window.QMPlanGuard = { getPlan, allow }; })();
