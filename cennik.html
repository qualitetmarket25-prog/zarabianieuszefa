(function(){ const PLAN_KEY = 'qm_plan_v1'; const MARGIN_KEY = 'qm_store_margin_pct'; const PRODUCTS_KEY = 'qm_products_by_supplier_v1'; const STORES_KEY = 'qm_stores_v1'; const ACTIVE_STORE_KEY = 'qm_active_store_v1'; const planMargins = { basic:18, pro:28, elite:38 }; const planPrices = { basic:79, pro:149, elite:299 }; function safeParse(key, fallback){ try{ return JSON.parse(localStorage.getItem(key) || JSON.stringify(fallback)); } catch(e){ return fallback; } } function save(key, value){ localStorage.setItem(key, JSON.stringify(value)); } function getPlan(){ return (localStorage.getItem(PLAN_KEY)||'basic').toLowerCase(); } function setPlan(plan){ const normalized = ['basic','pro','elite'].includes((plan||'').toLowerCase()) ? plan.toLowerCase() : 'basic'; localStorage.setItem(PLAN_KEY, normalized); const margin = planMargins[normalized] || 18; localStorage.setItem(MARGIN_KEY, String(margin)); localStorage.setItem('qm_subscription_v1', JSON.stringify({ plan: normalized, price: planPrices[normalized], status:'active', interval:'monthly', updatedAt: new Date().toISOString() })); applyMarginToStores(margin); applyMarginToProducts(); } function round99(value){ if(!isFinite(value)) return 0; const floored = Math.floor(value); return Math.max(0.99, Number((floored + 0.99).toFixed(2))); } function marginForPrice(base, plan){ const baseMargin = planMargins[plan] || 18; if(base <= 20) return baseMargin + 16; if(base <= 50) return baseMargin + 10; if(base <= 120) return baseMargin + 6; if(base <= 250) return baseMargin + 2; if(base <= 500) return baseMargin; return Math.max(12, baseMargin - 5); } function applyMarginToProducts(){ const plan = getPlan(); const products = safeParse(PRODUCTS_KEY, []); if(!Array.isArray(products)) return []; const next = products.map((p, idx) => { const base = Number(p.basePrice ?? p.cost ?? p.price ?? 0); const pct = marginForPrice(base, plan); const sale = round99(base * (1 + pct/100)); return { ...p, id: p.id || `prod_${idx+1}`, basePrice: Number(base.toFixed(2)), marginPct: pct, price: sale }; }); save(PRODUCTS_KEY, next); return next; } function applyMarginToStores(margin){ const stores = safeParse(STORES_KEY, []); const active = localStorage.getItem(ACTIVE_STORE_KEY) || ''; if(Array.isArray(stores) && stores.length){ const next = stores.map((s, idx) => { const isActive = s.slug === active || (!active && idx === 0); return isActive ? { ...s, marginPct: margin, updatedAt: new Date().toISOString() } : s; }); save(STORES_KEY, next); } } window.QMAutoMargin = { setPlan, applyMarginToProducts, getPlan, planMargins }; if(!localStorage.getItem(MARGIN_KEY)) localStorage.setItem(MARGIN_KEY, '18'); })();
