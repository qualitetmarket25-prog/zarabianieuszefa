(function(){ const PRODUCTS_KEY = 'qm_products_by_supplier_v1'; const MARGIN_KEY = 'qm_store_margin_pct'; const SUB_KEY = 'qm_subscription_v1'; function read(key, fallback){ try { return JSON.parse(localStorage.getItem(key)) ?? fallback; } catch(e){ return fallback; } } function write(key, value){ localStorage.setItem(key, JSON.stringify(value)); } function getMargin(){ const fromStore = Number(read(MARGIN_KEY, 0)); if (fromStore > 0) return fromStore; const planMargin = Number(read(SUB_KEY, {marginPct:18}).marginPct || 18); return planMargin; } function smartMargin(base, planMargin){ if (base < 20) return Math.max(planMargin, 35); if (base < 60) return Math.max(planMargin, 28); if (base < 150) return Math.max(planMargin, 22); if (base < 400) return Math.max(planMargin, 18); return Math.max(planMargin, 15); } function roundPrice(v){ const floor = Math.floor(Math.max(1, v)); return Number((floor + 0.99).toFixed(2)); } function normalizeBase(p){ const base = Number(p.sourcePrice ?? p.supplierPrice ?? p.basePrice ?? p.originalPrice ?? p.price ?? 0); return Number.isFinite(base) ? base : 0; } function productImage(name){ const n = encodeURIComponent(name || 'Produkt'); return `https://placehold.co/600x600/0f172a/e2e8f0?text=${n}`; } function enrichProduct(p, fallbackSupplier='AliExpress'){ const base = normalizeBase(p); const planMargin = getMargin(); const effective = smartMargin(base, planMargin); const finalPrice = roundPrice(base * (1 + effective / 100)); const compareAt = roundPrice(finalPrice * 1.18); return { ...p, supplier: p.supplier || fallbackSupplier, sourcePrice: base, supplierPrice: base, marginPct: effective, price: finalPrice, compareAtPrice: compareAt, img: p.img || p.image || productImage(p.name), _qm_margin_applied: true }; } function seedDemoIfEmpty(){ const curr = read(PRODUCTS_KEY, []); if (Array.isArray(curr) && curr.length) return curr; const demo = [ {name:'Organizer biurka premium', price:24.5, supplier:'CJ Dropshipping'}, {name:'Lampka LED do setupu', price:39.9, supplier:'AliExpress'}, {name:'Uchwyt samochodowy MagSafe', price:31.2, supplier:'Banggood'}, {name:'Plecak miejski smart', price:67.0, supplier:'EPROLO'}, {name:'Mata fitness premium', price:58.5, supplier:'VidaXL'}, {name:'Słuchawki bezprzewodowe', price:82.0, supplier:'AliExpress'} ].map(enrichProduct); write(PRODUCTS_KEY, demo); return demo; } function applyToCatalog(){ const existing = seedDemoIfEmpty(); const next = existing.map(p => enrichProduct(p, p.supplier || 'AliExpress')); write(PRODUCTS_KEY, next); return next; } document.addEventListener('DOMContentLoaded',()=>{ document.querySelectorAll('[data-apply-margin]').forEach(btn=>{ btn.addEventListener('click', ()=>{ applyToCatalog(); alert('Gotowe. Automatyczna marża została ustawiona na wszystkich produktach.'); }); }); }); window.QMAutoMargin = { getMargin, applyToCatalog, enrichProduct, seedDemoIfEmpty }; })();
